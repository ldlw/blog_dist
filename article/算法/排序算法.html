const arr = [11, 2, 4, 3, 6, 87]
/**
 *
 * @param {number[]} arr
 * @returns {number[]}
 */
function quickSort(arr) {
	if (arr.length < 2) {
		return arr
	}
	let pivot = arr[0]
	let less = arr.slice(1).filter(item => item <= pivot)
	let greater = arr.slice(1).filter(item => item > pivot)
	return quickSort(less).concat([pivot]).concat(quickSort(greater))
}
